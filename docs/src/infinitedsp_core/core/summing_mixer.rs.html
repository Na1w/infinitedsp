11/// Sums multiple audio signals together, with optional gain and soft clipping.
12///
13/// Useful for mixing multiple voices or signals.
14pub struct SummingMixer<C: ChannelConfig> {
15    inputs: Vec<Box<dyn FrameProcessor<C> + Send>>,
16    gain: AudioParam,
17    soft_clip: bool,
18    temp_buffer: Vec<f32>,
19    gain_buffer: Vec<f32>,
20}
21
22impl<C: ChannelConfig> SummingMixer<C> {
23    /// Creates a new SummingMixer with the given inputs.
24    pub fn new(inputs: Vec<Box<dyn FrameProcessor<C> + Send>>) -> Self {
25        SummingMixer {
26            inputs,
27            gain: AudioParam::Static(1.0),
28            soft_clip: false,
29            temp_buffer: Vec::new(),
30            gain_buffer: Vec::new(),
31        }
32    }
33
34    /// Sets the output gain.
35    pub fn set_gain(&mut self, gain: AudioParam) {
36        self.gain = gain;
37    }
38
39    /// Enables or disables soft clipping (tanh) on the output.
40    pub fn set_soft_clip(&mut self, enabled: bool) {
41        self.soft_clip = enabled;
42    }
43
44    /// Builder method to set gain.
45    pub fn with_gain(mut self, gain: AudioParam) -> Self {
46        self.gain = gain;
47        self
48    }
49
50    /// Builder method to enable soft clipping.
51    pub fn with_soft_clip(mut self, enabled: bool) -> Self {
52        self.soft_clip = enabled;
53        self
54    }
55}

57impl<C: ChannelConfig> FrameProcessor<C> for SummingMixer<C> {
58    fn process(&mut self, buffer: &mut [f32], sample_index: u64) {
59        if self.inputs.is_empty() {
60            buffer.fill(0.0);
61            return;
62        }
63
64        self.inputs[0].process(buffer, sample_index);
65
66        if self.inputs.len() > 1 {
67            if self.temp_buffer.len() < buffer.len() {
68                self.temp_buffer.resize(buffer.len(), 0.0);
69            }
70
71            let len = buffer.len();
72            let temp_slice = &mut self.temp_buffer[0..len];
73
74            for input in &mut self.inputs[1..] {
75                input.process(temp_slice, sample_index);
76
77                let (buf_chunks, buf_rem) = buffer.as_chunks_mut::<4>();
78                let (temp_chunks, temp_rem) = temp_slice.as_chunks::<4>();
79
80                for (buf_c, temp_c) in buf_chunks.iter_mut().zip(temp_chunks.iter()) {
81                    let buf_v = f32x4::from(*buf_c);
82                    let temp_v = f32x4::from(*temp_c);
83                    let res = buf_v + temp_v;
84                    *buf_c = res.to_array();
85                }
86
87                for (buf_s, temp_s) in buf_rem.iter_mut().zip(temp_rem.iter()) {
88                    *buf_s += *temp_s;
89                }
90            }
91        }
92
93        let constant_gain = self.gain.get_constant();
94        let skip_processing = !self.soft_clip && constant_gain == Some(1.0);
95
96        if !skip_processing {
97            let channels = C::num_channels();
98            let frames = buffer.len() / channels;
99
100            if self.gain_buffer.len() < frames {
101                self.gain_buffer.resize(frames, 0.0);
102            }
103
104            let gain_slice = &mut self.gain_buffer[0..frames];
105            self.gain.process(gain_slice, sample_index);
106
107            // Apply gain (and soft clip)
108            // We need to iterate samples and map them to the correct gain frame
109
110            for (i, sample) in buffer.iter_mut().enumerate() {
111                let frame_idx = i / channels;
112                let g = gain_slice[frame_idx];
113
114                let mut val = *sample * g;
115
116                if self.soft_clip {
117                    val = libm::tanhf(val);
118                }
119                *sample = val;
120            }
121        }
122    }
123
124    fn set_sample_rate(&mut self, sample_rate: f32) {
125        for input in &mut self.inputs {
126            input.set_sample_rate(sample_rate);
127        }
128        self.gain.set_sample_rate(sample_rate);
129    }
130
131    fn reset(&mut self) {
132        for input in &mut self.inputs {
133            input.reset();
134        }
135    }
136
137    fn latency_samples(&self) -> u32 {
138        self.inputs
139            .iter()
140            .map(|input| input.latency_samples())
141            .max()
142            .unwrap_or(0)
143    }
144
145    fn name(&self) -> &str {
146        "SummingMixer"
147    }
148
149    fn visualize(&self, indent: usize) -> String {
150        #[cfg(feature = "debug_visualize")]
151        {
152            let mut output = String::new();
153            let spaces = " ".repeat(indent);
154            let child_indent = indent + 2;
155
156            output.push_str(&format!("{}SummingMixer\n", spaces));
157
158            for (i, input) in self.inputs.iter().enumerate() {
159                output.push_str(&format!("{}Input {}:\n", " ".repeat(child_indent), i + 1));
160                output.push_str(&input.visualize(child_indent + 2));
161            }
162
163            output
164        }
165        #[cfg(not(feature = "debug_visualize"))]
166        {
167            let _ = indent;
168            String::new()
169        }
170    }
