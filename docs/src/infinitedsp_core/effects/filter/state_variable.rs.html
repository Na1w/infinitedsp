7/// Prewarp tangent for the TPT/ZDF bilinear transform.
8///
9/// Exact `libm::tanf` by default.
10#[cfg(not(feature = "perf-approximations"))]
11#[inline]
12fn prewarp_tan(x: f32) -> f32 {
13    libm::tanf(x)
14}
15
16/// Prewarp tangent — Padé[3/2] approximation `tan(x) ≈ x(15 - x²)/(15 - 6x²)`.
17///
18/// Matches `tan` through the x⁵ term: <0.2% error for x ≤ ~0.64 (covers all
19/// audio cutoffs up to ~9.8 kHz at 48 kHz), rising to ~10% only near the
20/// `0.49·fs` clamp. The denominator stays positive for x < π/2, and the SVF
21/// clamps the cutoff below that, so there is no pole. Enabled by the
22/// `perf-approximations` feature.
23#[cfg(feature = "perf-approximations")]
24#[inline]
25fn prewarp_tan(x: f32) -> f32 {
26    let x2 = x * x;
27    x * (15.0 - x2) / (15.0 - 6.0 * x2)
28}

30/// The output type of the State Variable Filter.
31#[derive(Clone, Copy)]
32pub enum SvfType {
33    LowPass,
34    HighPass,
35    BandPass,
36    Notch,
37    Peak,
38}

40/// A State Variable Filter (SVF).
41///
42/// A stable and versatile filter that provides simultaneous low-pass, high-pass, band-pass and notch outputs.
43/// This implementation uses the TPT (Topology Preserving Transform) / ZDF (Zero Delay Feedback) method
44/// for excellent stability and response across the frequency range.
45pub struct StateVariableFilter {
46    filter_type: SvfType,
47    cutoff: AudioParam,
48    resonance: AudioParam,
49    sample_rate: f32,
50    s1: f32,
51    s2: f32,
52
53    last_cutoff: f32,
54    last_res: f32,
55    g: f32,
56    k: f32,
57    // Per-sample-invariant quantities derived from g/k, cached behind the same
58    // change guard as g/k so the per-sample body needs no division.
59    denom: f32,    // 1 / (1 + g*(g+k))
60    g_plus_k: f32, // g + k
61    two_g: f32,    // 2*g
62
63    cutoff_buffer: Vec<f32>,
64    res_buffer: Vec<f32>,
65}

67impl StateVariableFilter {
68    /// Creates a new StateVariableFilter.
69    ///
70    /// # Arguments
71    /// * `filter_type` - The output type.

73    /// * `resonance` - Resonance (Q). 0.0 to 1.0 (or higher for self-oscillation).
74    pub fn new(filter_type: SvfType, cutoff: AudioParam, resonance: AudioParam) -> Self {
75        StateVariableFilter {

77            cutoff,
78            resonance,

81            s2: 0.0,
82            last_cutoff: -1.0,
83            last_res: -1.0,
84            g: 0.0,
85            k: 0.0,
86            denom: 0.0,
87            g_plus_k: 0.0,
88            two_g: 0.0,
89            cutoff_buffer: Vec::with_capacity(128),
90            res_buffer: Vec::with_capacity(128),
91        }
92    }
93
94    /// Sets the filter type.
95    pub fn set_type(&mut self, filter_type: SvfType) {
96        self.filter_type = filter_type;
97    }
98
99    /// Sets the cutoff frequency parameter.
100    pub fn set_cutoff(&mut self, cutoff: AudioParam) {
101        self.cutoff = cutoff;

109    /// Processes a single sample through the filter.
110    #[inline(always)]
111    pub fn tick(&mut self, input: f32, cutoff_hz: f32, res: f32) -> f32 {
112        if (cutoff_hz - self.last_cutoff).abs() > 0.001 || (res - self.last_res).abs() > 0.001 {
113            self.g = prewarp_tan(
114                (PI / self.sample_rate) * cutoff_hz.clamp(10.0, self.sample_rate * 0.49),
115            );
116            self.k = 1.0 / res.max(0.01);
117            // Recompute the g/k-derived constants only when g/k change.
118            self.g_plus_k = self.g + self.k;
119            self.two_g = 2.0 * self.g;
120            self.denom = 1.0 / (1.0 + self.g * self.g_plus_k);
121            self.last_cutoff = cutoff_hz;
122            self.last_res = res;
123        }
124
125        let hp = (input - self.s1 * self.g_plus_k - self.s2) * self.denom;
126        let bp = self.g * hp + self.s1;
127        let lp = self.g * bp + self.s2;
128
129        self.s1 += self.two_g * hp;
130        self.s2 += self.two_g * bp;
131
132        match self.filter_type {
133            SvfType::LowPass => lp,
134            SvfType::HighPass => hp,
135            SvfType::BandPass => bp,
136            SvfType::Notch => hp + lp,
137            SvfType::Peak => lp - hp,
138        }
139    }
140}

142impl FrameProcessor<Mono> for StateVariableFilter {
143    fn process(&mut self, buffer: &mut [f32], sample_index: u64) {
144        let len = buffer.len();
145        if self.cutoff_buffer.len() < len {
146            self.cutoff_buffer.resize(len, 0.0);
147        }
148        if self.res_buffer.len() < len {
149            self.res_buffer.resize(len, 0.0);
150        }
151
152        self.cutoff
153            .process(&mut self.cutoff_buffer[0..len], sample_index);
154        self.resonance
155            .process(&mut self.res_buffer[0..len], sample_index);
156
157        for (i, sample) in buffer.iter_mut().enumerate() {
158            *sample = self.tick(*sample, self.cutoff_buffer[i], self.res_buffer[i]);
159        }
160    }
161
162    fn set_sample_rate(&mut self, sample_rate: f32) {
163        self.sample_rate = sample_rate;
164        self.cutoff.set_sample_rate(sample_rate);
165        self.resonance.set_sample_rate(sample_rate);
166        self.last_cutoff = -1.0;
167    }
168
169    fn reset(&mut self) {
170        self.s1 = 0.0;
171        self.s2 = 0.0;
172    }
173
174    #[cfg(feature = "debug_visualize")]
175    fn name(&self) -> &str {
176        match self.filter_type {
177            SvfType::LowPass => "SVF (LowPass)",
178            SvfType::HighPass => "SVF (HighPass)",
179            SvfType::BandPass => "SVF (BandPass)",
180            SvfType::Notch => "SVF (Notch)",
181            SvfType::Peak => "SVF (Peak)",
182        }
183    }
184}
