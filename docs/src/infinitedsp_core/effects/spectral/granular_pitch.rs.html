7/// A granular pitch shifter.
8///
9/// Shifts pitch by resampling grains of audio.
10pub struct GranularPitchShift {
11    buffer: Vec<f32>,
12    write_ptr: usize,
13    phasor: f32,
14    window_size: f32,
15    semitones: AudioParam,
16    pitch_factor: f32,
17    window_ms: f32,
18    sample_rate: f32,
19    semitones_buffer: Vec<f32>,
20
21    last_semitones_bits: u32,
22}
23
24impl GranularPitchShift {
25    /// Creates a new GranularPitchShift.
26    ///
27    /// # Arguments
28    /// * `window_ms` - Grain size in milliseconds.
29    /// * `semitones` - Pitch shift amount in semitones.
30    pub fn new(window_ms: f32, semitones: AudioParam) -> Self {
31        let sample_rate = 44100.0;
32        let window_size = window_ms * sample_rate / 1000.0;
33        let buffer_size = (sample_rate * 0.5) as usize;
34
35        GranularPitchShift {
36            buffer: vec![0.0; buffer_size],
37            write_ptr: 0,
38            phasor: 0.0,
39            window_size,
40            semitones,
41            pitch_factor: 1.0,
42            window_ms,
43            sample_rate,
44            semitones_buffer: Vec::new(),
45            last_semitones_bits: u32::MAX,
46        }
47    }
48
49    /// Sets the pitch shift amount in semitones.
50    pub fn set_semitones(&mut self, semitones: AudioParam) {
51        self.semitones = semitones;
52    }
53}
54
55impl FrameProcessor<Mono> for GranularPitchShift {
56    fn process(&mut self, buffer: &mut [f32], sample_index: u64) {
57        let len = self.buffer.len() as f32;
58
59        if self.semitones_buffer.len() < buffer.len() {
60            self.semitones_buffer.resize(buffer.len(), 0.0);
61        }
62        self.semitones
63            .process(&mut self.semitones_buffer, sample_index);
64
65        for (i, sample) in buffer.iter_mut().enumerate() {
66            let semitones = self.semitones_buffer[i];
67
68            let semitones_bits = semitones.to_bits();
69            if semitones_bits != self.last_semitones_bits {
70                self.pitch_factor = libm::powf(2.0, semitones / 12.0);
71                self.last_semitones_bits = semitones_bits;
72            }
73
74            let inc = 1.0 - self.pitch_factor;
75
76            let input = *sample;
77            self.buffer[self.write_ptr] = input;
78
79            self.phasor += inc;
80            if self.phasor >= self.window_size {
81                self.phasor -= self.window_size;
82            } else if self.phasor < 0.0 {
83                self.phasor += self.window_size;
84            }
85
86            let delay1 = self.phasor;
87            let r1 = (self.write_ptr as f32 - delay1 + len) % len;
88            let val1 = self.buffer[r1 as usize];
89
90            let mut delay2 = self.phasor + self.window_size * 0.5;
91            if delay2 >= self.window_size {
92                delay2 -= self.window_size;
93            }
94            let r2 = (self.write_ptr as f32 - delay2 + len) % len;
95            let val2 = self.buffer[r2 as usize];
96
97            let x1 = delay1 / self.window_size;
98            let gain1 = if x1 < 0.5 { 2.0 * x1 } else { 2.0 * (1.0 - x1) };
99
100            let x2 = delay2 / self.window_size;
101            let gain2 = if x2 < 0.5 { 2.0 * x2 } else { 2.0 * (1.0 - x2) };
102
103            *sample = val1 * gain1 + val2 * gain2;
104
105            self.write_ptr = (self.write_ptr + 1) % self.buffer.len();
106        }
107    }
108
109    fn set_sample_rate(&mut self, sample_rate: f32) {
110        self.sample_rate = sample_rate;
111        self.semitones.set_sample_rate(sample_rate);
112        self.window_size = self.window_ms * sample_rate / 1000.0;
113        self.last_semitones_bits = u32::MAX;
114
115        let needed = (sample_rate * 0.5) as usize;
116        if needed > self.buffer.len() {
117            self.buffer.resize(needed, 0.0);
118        }
119    }
120
121    fn reset(&mut self) {
122        self.buffer.fill(0.0);
123        self.write_ptr = 0;
124        self.phasor = 0.0;
125    }
126
127    #[cfg(feature = "debug_visualize")]
128    fn name(&self) -> &str {
129        "GranularPitchShift"
130    }
131}
