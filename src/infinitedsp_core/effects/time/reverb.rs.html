8/// 4 parallel Comb filters (SIMD friendly... hopefully).
9struct Comb4 {
10    buffers: [Vec<f32>; 4],
11    pos: [usize; 4],
12
13    feedback: f32x4,
14    damp: f32x4,
15    damp_inv: f32x4,
16    filter_state: f32x4,
17}
18
19impl Comb4 {
20    fn new(sizes: [usize; 4], feedback: f32, damp: f32) -> Self {
21        assert!(
22            !sizes.iter().any(|&s| s < 1),
23            "Comb4: All filters must be at least 1 unit long."
24        );
25        Comb4 {
26            buffers: [
27                vec![0.0; sizes[0]],
28                vec![0.0; sizes[1]],
29                vec![0.0; sizes[2]],
30                vec![0.0; sizes[3]],
31            ],
32            pos: [0; 4],
33            feedback: f32x4::splat(feedback),
34            damp: f32x4::splat(damp),
35            damp_inv: f32x4::splat(1.0 - damp),
36            filter_state: f32x4::ZERO,
37        }
38    }
39
40    fn set_params(&mut self, feedback: f32, damp: f32, damp_inv: f32) {
41        self.feedback = f32x4::splat(feedback);
42        self.damp = f32x4::splat(damp);
43        self.damp_inv = f32x4::splat(damp_inv);
44    }
45
46    fn process(&mut self, input: f32) -> f32 {
47        let input_vec = f32x4::splat(input);
48
49        // SAFETY: pos can never be outside the bounds.
50        let delayed = unsafe {
51            let d0 = *self.buffers[0].get_unchecked(self.pos[0]);
52            let d1 = *self.buffers[1].get_unchecked(self.pos[1]);
53            let d2 = *self.buffers[2].get_unchecked(self.pos[2]);
54            let d3 = *self.buffers[3].get_unchecked(self.pos[3]);
55            f32x4::new([d0, d1, d2, d3])
56        };
57
58        let new_input = input_vec + self.filter_state * self.feedback;
59        self.filter_state = delayed * self.damp_inv + self.filter_state * self.damp;
60
61        let to_write = new_input.to_array();
62
63        // SAFETY: pos can never be outside the bounds.
64        unsafe {
65            *self.buffers[0].get_unchecked_mut(self.pos[0]) = to_write[0];
66            *self.buffers[1].get_unchecked_mut(self.pos[1]) = to_write[1];
67            *self.buffers[2].get_unchecked_mut(self.pos[2]) = to_write[2];
68            *self.buffers[3].get_unchecked_mut(self.pos[3]) = to_write[3];
69        };
70
71        for i in 0..4 {
72            self.pos[i] += 1;
73            if self.pos[i] >= self.buffers[i].len() {
74                self.pos[i] = 0;
75            }
76        }
77
78        delayed.reduce_add()
79    }
80
81    fn reset(&mut self) {
82        for buf in &mut self.buffers {
83            buf.fill(0.0);
84        }
85        self.pos = [0; 4];
86        self.filter_state = f32x4::ZERO;
87    }
88}

90struct Allpass {
91    buffer: Vec<f32>,
92    pos: usize,
93    feedback: f32,
94}
95
96impl Allpass {
97    fn new(size: usize) -> Self {
98        assert!(size > 0, "Allpass: Length must be at least one unit.");
99        Allpass {
100            buffer: vec![0.0; size],
101            pos: 0,
102            feedback: 0.5,
103        }
104    }
105
106    fn process(&mut self, input: f32) -> f32 {
107        let len = self.buffer.len();
108        // SAFETY: pos can never be outside the bounds.
109        let delayed = unsafe { *self.buffer.get_unchecked(self.pos) };
110        let output = -input + delayed;
111        let to_store = input + output * self.feedback;
112        // SAFETY: pos can never be outside the bounds.
113        unsafe { *self.buffer.get_unchecked_mut(self.pos) = to_store };
114
115        self.pos += 1;
116        if self.pos >= len {
117            self.pos = 0;
118        }
119        output
120    }
121
122    fn reset(&mut self) {
123        self.buffer.fill(0.0);
124        self.pos = 0;
125    }
126}

128pub struct Reverb {
129    combs_l: [Comb4; 2],
130    combs_r: [Comb4; 2],
131    allpasses_l: Vec<Allpass>,
132    allpasses_r: Vec<Allpass>,
133    room_size: AudioParam,
134    damping: AudioParam,

136}

138impl Reverb {
139    pub fn new() -> Self {
140        Self::new_with_params(AudioParam::Static(0.8), AudioParam::Static(0.2), 0)
141    }
142
143    pub fn new_with_seed(seed: usize) -> Self {
144        Self::new_with_params(AudioParam::Static(0.8), AudioParam::Static(0.2), seed)
145    }
146
147    pub fn new_with_params(room_size: AudioParam, damping: AudioParam, seed: usize) -> Self {
148        let comb_tuning = [1116, 1188, 1277, 1356, 1422, 1491, 1557, 1617];
149        let allpass_tuning = [556, 441, 341, 225];
150        let stereo_spread = 23;
151
152        let c1_l = [
153            comb_tuning[0] + seed,
154            comb_tuning[1] + seed,
155            comb_tuning[2] + seed,
156            comb_tuning[3] + seed,
157        ];
158        let c2_l = [
159            comb_tuning[4] + seed,
160            comb_tuning[5] + seed,
161            comb_tuning[6] + seed,
162            comb_tuning[7] + seed,
163        ];
164
165        let c1_r = [
166            comb_tuning[0] + stereo_spread + seed,
167            comb_tuning[1] + stereo_spread + seed,
168            comb_tuning[2] + stereo_spread + seed,
169            comb_tuning[3] + stereo_spread + seed,
170        ];
171        let c2_r = [
172            comb_tuning[4] + stereo_spread + seed,
173            comb_tuning[5] + stereo_spread + seed,
174            comb_tuning[6] + stereo_spread + seed,
175            comb_tuning[7] + stereo_spread + seed,
176        ];
177
178        let combs_l = [Comb4::new(c1_l, 0.8, 0.2), Comb4::new(c2_l, 0.8, 0.2)];
179        let combs_r = [Comb4::new(c1_r, 0.8, 0.2), Comb4::new(c2_r, 0.8, 0.2)];
180
181        let mut allpasses_l = Vec::new();
182        let mut allpasses_r = Vec::new();
183
184        for t in allpass_tuning {
185            allpasses_l.push(Allpass::new(t + seed));
186            allpasses_r.push(Allpass::new(t + stereo_spread + seed));
187        }
188
189        Reverb {
190            combs_l,
191            combs_r,
192            allpasses_l,
193            allpasses_r,
194            room_size,
195            damping,
196            sample_rate: 44100.0,
197        }
198    }

201        self.room_size = room_size;
202    }

205        self.damping = damping;

209impl FrameProcessor<Stereo> for Reverb {
210    fn process(&mut self, buffer: &mut [f32], sample_index: u64) {
211        let mut param_scratch = [0.0; 1];
212
213        self.room_size.process(&mut param_scratch, sample_index);
214        let rs = param_scratch[0] * 0.28 + 0.7;
215
216        self.damping.process(&mut param_scratch, sample_index);
217        let dp = param_scratch[0] * 0.4;
218        let dp_inv = 1.0 - dp;
219
220        for c in &mut self.combs_l {
221            c.set_params(rs, dp, dp_inv);
222        }
223        for c in &mut self.combs_r {
224            c.set_params(rs, dp, dp_inv);
225        }
226
227        for frame in buffer.chunks_mut(2) {
228            let input = (frame[0] + frame[1]) * 0.5 * 0.015;
229
230            let mut out_l = self.combs_l[0].process(input);
231            out_l += self.combs_l[1].process(input);
232
233            let mut out_r = self.combs_r[0].process(input);
234            out_r += self.combs_r[1].process(input);
235
236            for ap in &mut self.allpasses_l {
237                out_l = ap.process(out_l);
238            }
239            for ap in &mut self.allpasses_r {
240                out_r = ap.process(out_r);
241            }
242
243            frame[0] = out_l;
244            frame[1] = out_r;
245        }
246    }
247
248    fn set_sample_rate(&mut self, sample_rate: f32) {
249        self.sample_rate = sample_rate;
250        self.room_size.set_sample_rate(sample_rate);
251        self.damping.set_sample_rate(sample_rate);
252    }
253
254    fn reset(&mut self) {
255        for c in &mut self.combs_l {
256            c.reset();
257        }
258        for c in &mut self.combs_r {
259            c.reset();
260        }
261        for ap in &mut self.allpasses_l {
262            ap.reset();
263        }
264        for ap in &mut self.allpasses_r {
265            ap.reset();
266        }
267        self.room_size.reset();
268        self.damping.reset();
269    }
270
271    #[cfg(feature = "debug_visualize")]
272    fn name(&self) -> &str {
273        "Reverb (Schroeder)"
274    }
275}
