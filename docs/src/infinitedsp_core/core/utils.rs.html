1/// A lightweight random number generator.
2pub struct FastRng {
3    state: u32,
4}
5
6impl FastRng {
7    /// Creates a new FastRng with a given seed.
8    pub fn new(seed: u32) -> Self {
9        Self { state: seed }
10    }
11
12    /// Generates the next random `u32` from a state.
13    #[inline(always)]
14    pub fn next_u32_stateless(state: &mut u32) -> u32 {
15        *state = state.wrapping_mul(1103515245).wrapping_add(12345);
16        *state
17    }
18
19    /// Generates a random `f32` in the range [-1.0, 1.0] from a state.
20    #[inline(always)]
21    pub fn next_f32_bipolar_stateless(state: &mut u32) -> f32 {
22        let val = (Self::next_u32_stateless(state) >> 16) & 0x7FFF;
23        (val as f32 / 32768.0) * 2.0 - 1.0
24    }
25
26    /// Generates a random `f32` in the range [0.0, 1.0) from a state.
27    #[inline(always)]
28    pub fn next_f32_unipolar_stateless(state: &mut u32) -> f32 {
29        (Self::next_u32_stateless(state) as f32) / (u32::MAX as f32)
30    }
31
32    /// Generates the next random `u32`.
33    #[inline(always)]
34    pub fn next_u32(&mut self) -> u32 {
35        Self::next_u32_stateless(&mut self.state)
36    }
37
38    /// Generates a random `f32` in the range [-1.0, 1.0].
39    #[inline(always)]
40    pub fn next_f32_bipolar(&mut self) -> f32 {
41        Self::next_f32_bipolar_stateless(&mut self.state)
42    }
43
44    /// Generates a random `f32` in the range [0.0, 1.0).
45    #[inline(always)]
46    pub fn next_f32_unipolar(&mut self) -> f32 {
47        Self::next_f32_unipolar_stateless(&mut self.state)
48    }
49}
50
51impl Default for FastRng {
52    fn default() -> Self {
53        Self::new(12345)
54    }
55}
