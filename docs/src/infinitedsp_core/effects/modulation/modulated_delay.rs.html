8/// A modulated delay effect, used for Chorus and Flanger.
9pub struct ModulatedDelay {
10    buffer: Vec<f32>,
11    write_ptr: usize,
12
13    lfo_phase: f32,
14    lfo_inc: f32,
15    depth: AudioParam,
16    base_delay: f32,
17
18    feedback: AudioParam,
19    mix: AudioParam,
20    sample_rate: f32,
21
22    depth_buffer: Vec<f32>,
23    feedback_buffer: Vec<f32>,
24    mix_buffer: Vec<f32>,
25}

27impl ModulatedDelay {
28    /// Creates a new Chorus effect.
29    ///
30    /// Uses a longer delay time and moderate modulation depth.
31    pub fn new_chorus() -> Self {
32        let sample_rate = 44100.0;
33        let buffer_size = (sample_rate * 0.1) as usize;
34
35        ModulatedDelay {
36            buffer: vec![0.0; buffer_size],
37            write_ptr: 0,
38            lfo_phase: 0.0,
39            lfo_inc: 2.0 * PI * 1.5 / sample_rate,
40            depth: AudioParam::Static(0.002 * sample_rate),
41            base_delay: 0.015 * sample_rate,
42            feedback: AudioParam::Static(0.4),
43            mix: AudioParam::Static(0.5),
44            sample_rate,
45            depth_buffer: Vec::new(),
46            feedback_buffer: Vec::new(),
47            mix_buffer: Vec::new(),
48        }
49    }
50
51    /// Creates a new Flanger effect.
52    ///
53    /// Uses a short delay time and higher feedback.
54    pub fn new_flanger() -> Self {
55        let sample_rate = 44100.0;
56        let buffer_size = (sample_rate * 0.1) as usize;
57
58        ModulatedDelay {
59            buffer: vec![0.0; buffer_size],
60            write_ptr: 0,
61            lfo_phase: 0.0,
62            lfo_inc: 2.0 * PI * 0.5 / sample_rate,
63            depth: AudioParam::Static(0.005 * sample_rate),
64            base_delay: 0.005 * sample_rate,
65            feedback: AudioParam::Static(0.7),
66            mix: AudioParam::Static(0.5),
67            sample_rate,
68            depth_buffer: Vec::new(),
69            feedback_buffer: Vec::new(),
70            mix_buffer: Vec::new(),
71        }
72    }

90impl FrameProcessor<Mono> for ModulatedDelay {
91    fn process(&mut self, buffer: &mut [f32], sample_index: u64) {
92        let len = self.buffer.len();
93        let len_f = len as f32;
94        let block_size = buffer.len();
95
96        if self.depth_buffer.len() < block_size {
97            self.depth_buffer.resize(block_size, 0.0);
98        }
99        if self.feedback_buffer.len() < block_size {
100            self.feedback_buffer.resize(block_size, 0.0);
101        }
102        if self.mix_buffer.len() < block_size {
103            self.mix_buffer.resize(block_size, 0.0);
104        }
105
106        self.depth
107            .process(&mut self.depth_buffer[0..block_size], sample_index);
108        self.feedback
109            .process(&mut self.feedback_buffer[0..block_size], sample_index);
110        self.mix
111            .process(&mut self.mix_buffer[0..block_size], sample_index);
112
113        for (i, sample) in buffer.iter_mut().enumerate() {
114            let input = *sample;
115            let depth = self.depth_buffer[i];
116            let feedback = self.feedback_buffer[i];
117            let mix = self.mix_buffer[i];
118
119            self.lfo_phase += self.lfo_inc;
120            if self.lfo_phase > 2.0 * PI {
121                self.lfo_phase -= 2.0 * PI;
122            }
123
124            let lfo = libm::sinf(self.lfo_phase);
125            let current_delay = self.base_delay + lfo * depth;
126
127            let read_pos = (self.write_ptr as f32 - current_delay + len_f) % len_f;
128            let idx_a = read_pos as usize;
129            let idx_b = (idx_a + 1) % len;
130            let frac = read_pos - idx_a as f32;
131
132            let delayed = self.buffer[idx_a] * (1.0 - frac) + self.buffer[idx_b] * frac;
133
134            self.buffer[self.write_ptr] = input + delayed * feedback;
135
136            *sample = input * (1.0 - mix) + delayed * mix;
137
138            self.write_ptr = (self.write_ptr + 1) % len;
139        }
140    }
141
142    fn set_sample_rate(&mut self, sample_rate: f32) {
143        let old_sr = self.sample_rate;
144        self.sample_rate = sample_rate;
145        self.depth.set_sample_rate(sample_rate);
146        self.feedback.set_sample_rate(sample_rate);
147        self.mix.set_sample_rate(sample_rate);
148
149        self.lfo_inc = self.lfo_inc * old_sr / sample_rate;
150
151        self.base_delay = self.base_delay * sample_rate / old_sr;
152
153        let needed = (sample_rate * 0.1) as usize;
154        if needed > self.buffer.len() {
155            self.buffer.resize(needed, 0.0);
156        }
157    }
158
159    fn reset(&mut self) {
160        self.buffer.fill(0.0);
161        self.write_ptr = 0;
162        self.lfo_phase = 0.0;
163    }
164
165    #[cfg(feature = "debug_visualize")]
166    fn name(&self) -> &str {
167        "ModulatedDelay (Chorus/Flanger)"
168    }
169}
