1use crate::core::audio_param::AudioParam;
2use crate::core::channels::Stereo;
3use crate::FrameProcessor;
4use alloc::vec::Vec;
5
6/// A simple Mid/Side stereo widener.
7///
8/// Increases or decreases the stereo width by adjusting the Side channel level.
9/// Width > 1.0 increases width, < 1.0 decreases width (0.0 = mono).
10pub struct StereoWidener {
11    width: AudioParam,
12    width_buffer: Vec<f32>,
13}
14
15impl StereoWidener {
16    /// Creates a new StereoWidener.
17    ///
18    /// # Arguments
19    /// * `width` - Stereo width factor (1.0 = normal, 0.0 = mono, >1.0 = wide).
20    pub fn new(width: AudioParam) -> Self {
21        StereoWidener {
22            width,
23            width_buffer: Vec::new(),
24        }
25    }
26}
27
28impl FrameProcessor<Stereo> for StereoWidener {
29    fn process(&mut self, buffer: &mut [f32], sample_index: u64) {
30        let frames = buffer.len() / 2;
31        if self.width_buffer.len() < frames {
32            self.width_buffer.resize(frames, 0.0);
33        }
34
35        self.width
36            .process(&mut self.width_buffer[0..frames], sample_index);
37
38        for (i, frame) in buffer.chunks_mut(2).enumerate() {
39            if frame.len() < 2 {
40                break;
41            }
42            let width = self.width_buffer[i];
43
44            let l = frame[0];
45            let r = frame[1];
46
47            let mid = (l + r) * 0.5;
48            let side = (l - r) * 0.5;
49
50            let side_boosted = side * width;
51
52            frame[0] = mid + side_boosted;
53            frame[1] = mid - side_boosted;
54        }
55    }
56
57    fn set_sample_rate(&mut self, sample_rate: f32) {
58        self.width.set_sample_rate(sample_rate);
59    }
60
61    #[cfg(feature = "debug_visualize")]
62    fn name(&self) -> &str {
63        "StereoWidener"
64    }
65}
