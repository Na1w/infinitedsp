7struct Allpass {
8    zm1: f32,
9}
10
11impl Allpass {
12    fn new() -> Self {
13        Allpass { zm1: 0.0 }
14    }
15
16    fn process(&mut self, input: f32, a1: f32) -> f32 {
17        let y = input * -a1 + self.zm1;
18        self.zm1 = input + y * a1;
19        y
20    }
21
22    fn reset(&mut self) {
23        self.zm1 = 0.0;
24    }
25}

120impl FrameProcessor<Mono> for Phaser {
121    fn process(&mut self, buffer: &mut [f32], sample_index: u64) {
122        let len = buffer.len();
123        if self.rate_buffer.len() < len {
124            self.rate_buffer.resize(len, 0.0);
125        }
126        if self.min_freq_buffer.len() < len {
127            self.min_freq_buffer.resize(len, 0.0);
128        }
129        if self.max_freq_buffer.len() < len {
130            self.max_freq_buffer.resize(len, 0.0);
131        }
132        if self.feedback_buffer.len() < len {
133            self.feedback_buffer.resize(len, 0.0);
134        }
135        if self.mix_buffer.len() < len {
136            self.mix_buffer.resize(len, 0.0);
137        }
138
139        self.rate
140            .process(&mut self.rate_buffer[0..len], sample_index);
141        self.min_freq
142            .process(&mut self.min_freq_buffer[0..len], sample_index);
143        self.max_freq
144            .process(&mut self.max_freq_buffer[0..len], sample_index);
145        self.feedback
146            .process(&mut self.feedback_buffer[0..len], sample_index);
147        self.mix.process(&mut self.mix_buffer[0..len], sample_index);
148
149        for (i, sample) in buffer.iter_mut().enumerate() {
150            let rate = self.rate_buffer[i];
151            let min_freq = self.min_freq_buffer[i];
152            let max_freq = self.max_freq_buffer[i];
153            let feedback = self.feedback_buffer[i];
154            let mix = self.mix_buffer[i];
155
156            self.lfo_inc = 2.0 * PI * rate / self.sample_rate;
157
158            let input = *sample + self.last_sample * feedback;
159
160            self.lfo_phase += self.lfo_inc;
161            if self.lfo_phase > 2.0 * PI {
162                self.lfo_phase -= 2.0 * PI;
163            }
164
165            let lfo = (libm::sinf(self.lfo_phase) + 1.0) * 0.5;
166            let freq = min_freq + lfo * (max_freq - min_freq);
167
168            let w = 2.0 * PI * freq / self.sample_rate;
169            let tan = libm::tanf(w * 0.5);
170
171            let a1 = (1.0 - tan) / (1.0 + tan);
172
173            let mut out = input;
174            for filter in &mut self.filters {
175                out = filter.process(out, a1);
176            }
177
178            self.last_sample = out;
179            *sample = *sample * (1.0 - mix) + out * mix;
180        }
181    }
182
183    fn set_sample_rate(&mut self, sample_rate: f32) {
184        self.sample_rate = sample_rate;
185        self.rate.set_sample_rate(sample_rate);
186        self.min_freq.set_sample_rate(sample_rate);
187        self.max_freq.set_sample_rate(sample_rate);
188        self.feedback.set_sample_rate(sample_rate);
189        self.mix.set_sample_rate(sample_rate);
190    }
191
192    fn reset(&mut self) {
193        for filter in &mut self.filters {
194            filter.reset();
195        }
196        self.last_sample = 0.0;
197        self.lfo_phase = 0.0;
198    }
199
200    #[cfg(feature = "debug_visualize")]
201    fn name(&self) -> &str {
202        "Phaser"
203    }
204}
